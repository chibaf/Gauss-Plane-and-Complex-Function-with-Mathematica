    • (Local) In[8]:= a=Table[N[i/10] + I N[j/10],	(* I = Sqrt[-1] *)

  • Function, shows lattice
    • (Local) In[9]:= plt[clr_,x__]:=
			Show[Graphics[{Hue[clr],bl[x]}],

			AspectRatio->Automatic,Frame->True]
